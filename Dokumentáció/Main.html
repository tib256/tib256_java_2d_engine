-ütközés érzékelés(Collision),

import game.Menu;

import tib256_engine.GameNavigator;

import java.awt.Dimension;

import java.awt.event.KeyAdapter;
import java.awt.event.KeyEvent;
import java.awt.event.MouseEvent;
import java.awt.event.MouseListener;
import java.awt.event.MouseMotionAdapter;
import java.awt.event.MouseMotionListener;
import javax.swing.JComponent;
import javax.swing.JFrame;
import javax.swing.Timer;

/**
*
* @author Tibor
*/
public class Main extends JComponent implements ActionListener

{
 
  private String title;
  private int width;
  private int height;
 
  private Timer timer;
 
  private JFrame window;
 
  //Létrehozom a játék ablakaít
  private Game game = new Game();
  private Menu menu = new Menu();
  private Platform platform = new Platform() ;
 
public Main(String title,int width, int height)

{
   this.title = title;
   this.width = width;
   this.height = height;
   init();
   window.setVisible(true);
  
}

 
private void init()

{
   window = new JFrame(title);
   window.add(this);
   window.pack();
   window.setSize(new Dimension(width,height));
   window.setLocationRelativeTo(null);
   window.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
   window.setResizable(false);
  
  
  
  
window.addKeyListener(new KeyAdapter(){
   @Override
public void keyPressed(KeyEvent e)


switch(GameNavigator.gamestate){
    case GAME: game.keyPressed(e);
    break;
    case PLATFORM: platform.keyPressed(e);
    break;
    case MENU:
    break;
}
}

  
   @Override
public void keyReleased(KeyEvent e)


switch(GameNavigator.gamestate){
    case GAME: game.keyReleased(e);
    break;
    case PLATFORM: platform.keyReleased(e);
    break;
    case MENU:
    break;
}
}

  
});
  
window.addMouseListener(new MouseInputAdapter(){
  
   @Override
public void mousePressed(MouseEvent e)


switch(GameNavigator.gamestate){
    case GAME: game.mousePressed(e);
    break;
    case MENU: menu.mousePressed(e);
    break;
    case PLATFORM: platform.mousePressed(e);
    break;
}
}

  
});
  
window.addMouseMotionListener(new MouseMotionAdapter(){
   @Override
public void mouseMoved(MouseEvent e)


switch(GameNavigator.gamestate){
    case GAME: game.mouseMoved(e);
    break;
    case MENU: menu.mouseMoved(e);
    break;
    case PLATFORM: platform.mouseMoved(e);
    break;
}
}

  
   @Override
public void mouseDragged(MouseEvent e)

{
   
}

  
  
} );
  
  
   timer = new Timer(17, this);
   timer.start();
  
   //GameNavigator.gamestate=GameNavigator.GameStates.GAME;
}

 
 
  @Override
public void actionPerformed(ActionEvent ae)

{
switch(GameNavigator.gamestate){
   case GAME: game.update();
   break;
   case MENU: menu.update();
   break;
   case PLATFORM: platform.update();
   break;
  
}
  
}
   repaint();
}
  
   @Override
protected void paintComponent(Graphics g){


switch(GameNavigator.gamestate){
   case GAME: game.render(g);
   break;
   case MENU: menu.render(g);
   break;
   case PLATFORM: platform.render(g);
   break;
  
}
  
}
  
  
public String getTitle()

{
    return title;
}

  
public void setTitle(String title)

{
    this.title = title;
}

  
public int getWidth()

{
    return width;
}

  
public void setWidth(int width)

{
    this.width = width;
}

  
public int getHeight()

{
    return height;
}

  
public void setHeight(int height)

{
    this.height = height;
}

  
public static void main(String[] args)

{
    new Main("Tib256 engine Test", 800, 600);
}

  
}
