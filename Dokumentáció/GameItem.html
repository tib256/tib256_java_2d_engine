package tib256_engine;

import tib256_engine.Texture;
import java.awt.image.BufferedImage;

/**
*
* @author Tibor
*/
public class GameItem

{
  public String name="";
 
  public int x = 100;
  public int y = 100;
  public int width = 100;
  public int height = 100;
  public int sourceX = 0;
  public int sourceY = 0;
  public int destX = 100;
  public int destY = 100;
 
  public float velocity = 3;
  public float vx = 0;
  public float vy = 0;
  public float jumpForce = -11;
  public float gravity = 0.3f;
  public float friction = 0.96f;
  public float bounce = -0.3f;
 
  public int numberOfFrames = 5;
  public int currentFrame = 0;
  public int COLUMNS = 3;
  public boolean animend = false;
 
  public boolean mu = false;
  public boolean md = false;
  public boolean ml = false;
  public boolean mr = false;
  public boolean visible = true;
  public boolean isOnGround = true;
 
  public boolean moveable = false;
 
  public BufferedImage img;
 
  public String filename;
 
public void init()

{
   Texture t = new Texture("/images");
   img = t.load(filename);
}

 
public void updateAnimation()

{
   this.sourceX
   =(int)Math.round( Math.floor(this.currentFrame % this.COLUMNS) * this.width);
   this.sourceY
   = (int)Math.round( Math.floor(this.currentFrame / this.COLUMNS) * this.height);
  
   this.destX
   =(int)Math.round( Math.floor(this.currentFrame % this.COLUMNS) * this.width)+this.width;
   this.destY
   = (int)Math.round( Math.floor(this.currentFrame / this.COLUMNS) * this.height)+this.height;
}

 
public double centerX()

{
   return this.x + (this.width * 0.5);
}

 
  public double centerY()

 

{
   return this.y + (this.height * 0.5);
}

 
  public double halfWidth()

 

{
   return this.width * 0.5;
}

 
  public double halfHeight()

 

{
   return this.height * 0.5;
}

}
