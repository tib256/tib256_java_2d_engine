public void update()

{
  
if(!pause)

{
    move(kor1);
    if(!moveable)
    Collision.blockCircle(kor1,kor2);
    else Collision.blockCircle(kor2,kor1);
    // System.out.println(time);
   
    time++;
   
if(time%40==0)

{
     if(!bagoly.animend)
     bagoly.currentFrame++;
     else
     bagoly.currentFrame--;
    
     if(bagoly.currentFrame==bagoly.numberOfFrames) bagoly.animend=true;
     if(bagoly.currentFrame==0) bagoly.animend=false;
    
}

   
    bagoly.updateAnimation();
   
}

  
  
}

private void move(GameItem p)


if(p.md)

{
    p.vy=p.velocity;
}

  
if(p.mu)

{
    p.vy=-p.velocity;;
}

  
if(p.ml)

{
    p.vx=-p.velocity;;
}

  
if(p.mr)

{
    p.vx=p.velocity;
}

  
  
   p.x += p.vx;
  
   p.y += p.vy;
  
  
}

 
 
public void keyPressed(KeyEvent e)

{
   int key = e.getKeyCode();
  
if(key == KeyEvent.VK_DOWN)


for (GameItem p : gameItems)

{
     if(p.moveable) p.md = true;
    
}

}

  
if(key == KeyEvent.VK_UP)


for (GameItem p : gameItems)

{
     if(p.moveable) p.mu = true;
}

}

  
if(key == KeyEvent.VK_LEFT)


for (GameItem p : gameItems)

{
     if(p.moveable) p.ml = true;
}

}

  
if(key == KeyEvent.VK_RIGHT)


for (GameItem p : gameItems)

{
     if(p.moveable) p.mr = true;
}

}

  
if(key == KeyEvent.VK_E)

{
    moveable = true;
}

  
if(key == KeyEvent.VK_P)

{
    pause = true;
    panel_pause.visible=true;
    button_continue.visible=true;
    button_exit.visible=true;
}

  
if(key == KeyEvent.VK_ESCAPE)


if(esc)

{
     pause = true;
     panel_pause.visible=true;
     button_continue.visible=true;
     button_exit.visible=true;
     esc=false;
}

else

{
     button_exit.visible=false;
     button_continue.visible=false;
     panel_pause.visible=false;
     pause=false;
     esc=true;
}

}

  
if(key == KeyEvent.VK_SPACE)

{
    jump = true;
}

  
}

 
public void keyReleased(KeyEvent e)

{
   int key = e.getKeyCode();
  
if(key == KeyEvent.VK_DOWN)


for (GameItem p : gameItems)

{
     if(p.moveable) p.md = false;
     if(p.moveable) p.vy = 0;
}

}

  
if(key == KeyEvent.VK_UP)


for (GameItem p : gameItems)

{
     if(p.moveable) p.mu = false;
     if(p.moveable) p.vy = 0;
}

}

  
if(key == KeyEvent.VK_LEFT)


for (GameItem p : gameItems)

{
     if(p.moveable) p.ml = false;
     if(p.moveable) p.vx = 0;
}

}

  
if(key == KeyEvent.VK_RIGHT)


for (GameItem p : gameItems)

{
     if(p.moveable) p.mr = false;
     if(p.moveable) p.vx = 0;
}

}

  
if(key == KeyEvent.VK_E)

{
    moveable = false;
}

  
if(key == KeyEvent.VK_SPACE)

{
    jump = false;
}

  
}
